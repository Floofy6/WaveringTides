### Enhanced Skill XP Formula

```typescript
export class SkillService {
  // Improved XP calculation with diminishing returns at higher levels
  calculateXpGain(skill: Skill, elapsedTime: number): number {
    // Base XP per action
    const baseXp = skill.xpPerAction;
    
    // Actions per second (adjust for different skills)
    const actionsPerSecond = this.getActionsPerSecond(skill);
    
    // Calculate actions in elapsed time
    const actions = (elapsedTime / 1000) * actionsPerSecond;
    
    // Apply level scaling (slower at higher levels)
    const levelScaling = 1 + ((skill.level - 1) * 0.01);
    
    return Math.floor(baseXp * actions * levelScaling);
  }
  
  getActionsPerSecond(skill: Skill): number {
    // Different skills have different action speeds
    switch(skill.id) {
      case SKILL_IDS.WOODCUTTING: return 0.8;
      case SKILL_IDS.FISHING: return 0.6;
      case SKILL_IDS.MINING: return 0.7;
      // Other skills with appropriate rates
      default: return 0.5;
    }
  }
}
```

```typescript
      this.intervalId = null;
    }
  }
```

### Resource Generation with Variability

```typescript
export class ResourceGenerator {
  generateResources(skill: Skill, elapsedTime: number): Item[] {
    const resources: Item[] = [];
    const actionTime = this.getActionTime(skill);
    const totalActions = Math.floor(elapsedTime / actionTime);
    
    for (let i = 0; i < totalActions; i++) {
      const items = this.rollForResources(skill);
      resources.push(...items);
    }
    
    return resources;
  }
  
  rollForResources(skill: Skill): Item[] {
    const result: Item[] = [];
    
    switch (skill.id) {
      case SKILL_IDS.WOODCUTTING:
        if (Math.random() < this.getSuccessRate(skill)) {
          result.push({ ...ITEMS[ITEM_IDS.LOGS], quantity: 1 });
          
          // Chance for rare drops (bird nests, etc.)
          if (Math.random() < 0.01) {
            result.push({ ...ITEMS[ITEM_IDS.BIRD_NEST], quantity: 1 });
          }
        }
        break;
      // Other skills
    }
    
    return result;
  }
  
  getSuccessRate(skill: Skill): number {
    // Higher levels = higher success rate
    return Math.min(0.3 + (skill.level * 0.007), 0.95);
  }
  
  getActionTime(skill: Skill): number {
    // Base time per action in milliseconds
    const baseTime = 3000;
    // Reduction based on level (up to 50% faster at level 99)
    const levelReduction = (skill.level - 1) * 15;
    return Math.max(baseTime - levelReduction, 1500);
  }
}
```

```typescript
export class FishingService {
  rollForFish(player: Player, fishingSpot: FishingSpot): Item[] {
    const fishingSkill = player.skills[SKILL_IDS.FISHING];
    const fishingLevel = fishingSkill.level;
    
    // Check if player meets level requirements
    if (fishingLevel < fishingSpot.requiredLevel) {
      return [];
    }
    
    const catches: Item[] = [];
    const baseSuccessRate = 0.3 + (fishingLevel * 0.005);
    
    // Roll for each fish in the spot based on weighted probability
    const totalWeight = fishingSpot.fish.reduce((sum, fish) => sum + fish.weight, 0);
    
    if (Math.random() < baseSuccessRate) {
      // Determine which fish is caught
      const roll = Math.random() * totalWeight;
      let cumulativeWeight = 0;
      
      for (const fish of fishingSpot.fish) {
        cumulativeWeight += fish.weight;
        if (roll <= cumulativeWeight) {
          catches.push({ ...ITEMS[fish.itemId], quantity: 1 });
          
          // Add XP
          fishingSkill.addXp(fish.xp);
          break;
        }
      }
    }
    
    return catches;
  }
}
```

### Enhanced Combat System

```typescript
export class CombatService {
  // Combat tick function - called each combat interval
  processCombatTick(player: Player): void {
    if (!player.combat.isFighting || !player.combat.currentEnemy) {
      return;
    }
    
    const enemy = player.combat.currentEnemy;
    
    // Calculate attack interval based on weapon speed
    const playerAttackTime = this.getAttackInterval(player);
    const enemyAttackTime = enemy.attackSpeed;
    
    // Update attack timers
    player.combat.playerAttackTimer += COMBAT_TICK_MS;
    player.combat.enemyAttackTimer += COMBAT_TICK_MS;
    
    // Process player attack
    if (player.combat.playerAttackTimer >= playerAttackTime) {
      player.combat.playerAttackTimer -= playerAttackTime;
      this.processPlayerAttack(player, enemy);
    }
    
    // Process enemy attack if enemy is still alive
    if (enemy.health > 0 && player.combat.enemyAttackTimer >= enemyAttackTime) {
      player.combat.enemyAttackTimer -= enemyAttackTime;
      this.processEnemyAttack(player, enemy);
    }
    
    // Check if enemy is defeated
    if (enemy.health <= 0) {
      this.handleEnemyDefeat(player, enemy);
    }
    
    // Check if player is defeated
    if (player.getCurrentHitpoints() <= 0) {
      this.handlePlayerDefeat(player);
    }
  }
  
  processPlayerAttack(player: Player, enemy: Enemy): void {
    // Calculate hit chance
    const accuracy = this.calculatePlayerAccuracy(player);
    const enemyEvasion = this.calculateEnemyEvasion(enemy);
    
    const hitChance = accuracy / (accuracy + enemyEvasion);
    
    if (Math.random() <= hitChance) {
      // Calculate damage
      const damage = this.calculatePlayerDamage(player);
      
      // Apply damage to enemy
      enemy.health = Math.max(0, enemy.health - damage);
      
      // Award combat XP
      player.addXpToSkill(SKILL_IDS.ATTACK, damage * 4);
      player.addXpToSkill(SKILL_IDS.STRENGTH, damage * 4);
      player.addXpToSkill(SKILL_IDS.HITPOINTS, damage * 1.33);
    } else {
      // Miss - no damage
    }
  }
  
  calculatePlayerDamage(player: Player): number {
    const strengthLevel = player.skills[SKILL_IDS.STRENGTH].level;
    const weaponDamage = player.equipment.weapon?.stats?.strengthBonus || 1;
    
    // Base damage calculation
    const maxHit = Math.floor(1.3 + (strengthLevel / 10) + (weaponDamage / 5));
    
    // Random roll between 0 and max hit
    return Math.floor(Math.random() * (maxHit + 1));
  }
  
  getAttackInterval(player: Player): number {
    // Base attack interval in milliseconds
    const baseInterval = 2400;
    
    // Get weapon speed modifier (if any)
    const weapon = player.equipment.weapon;
    const speedModifier = weapon?.stats?.speedBonus || 0;
    
    return baseInterval - (speedModifier * 200);
  }
}
```

```typescript
export class OfflineProgressService {
  applyOfflineProgress(player: Player): Player {
    const now = Date.now();
    const lastUpdate = player.lastUpdate;
    const elapsedTime = now - lastUpdate;
    
    // Cap offline progress (max 24 hours)
    const maxOfflineTime = 24 * 60 * 60 * 1000;
    const effectiveElapsedTime = Math.min(elapsedTime, maxOfflineTime);
    
    console.log(`Applying offline progress for ${effectiveElapsedTime}ms`);
    
    // Apply skill progress
    Object.values(player.skills).forEach(skill => {
      if (skill.isActive) {
        // Slightly reduced rate for offline progress (80%)
        const offlineRate = 0.8;
        const skillService = new SkillService(new InMemoryItemRepository());
        
        // Calculate XP gain
        const xpGain = skillService.calculateXpGain(skill, effectiveElapsedTime) * offlineRate;
        skill.addXp(xpGain);
        
        // Generate resources
        this.generateOfflineResources(player, skill, effectiveElapsedTime * offlineRate);
      }
    });
    
    // Combat is not processed offline in MVP
    player.combat.isFighting = false;
    player.combat.currentEnemy = undefined;
    
    // Update timestamp
    player.lastUpdate = now;
    
    return player;
  }
  
  generateOfflineResources(player: Player, skill: Skill, elapsedTime: number): void {
    // Similar to regular resource generation, but simplified and capped
    const resourceGenerator = new ResourceGenerator();
    const resources = resourceGenerator.generateResources(skill, elapsedTime);
    
    // Add resources to inventory
    resources.forEach(item => {
      player.inventory.addItem(item);
    });
  }
}
```

```typescript
export class Skill {
  addXp(xp: number): void {
    this.xp += xp;
    this.updateLevel();
  }

  updateLevel(): void {
    while (this.level < 99 && this.xp >= this.getXpRequiredForLevel(this.level + 1)) {
      this.level++;
    }
  }

  getXpRequiredForLevel(level: number): number {
    // Exponential formula creates increasing difficulty curve
    let points = 0;
    let output = 0;
    
    for (let lvl = 1; lvl <= level; lvl++) {
      points += Math.floor(lvl + 300 * Math.pow(2, lvl / 7));
      if (lvl >= 2) {
        output = Math.floor(points / 4);
      }
    }
    
    return output;
  }
}
```

### 7. Persistence and Offline Progress

Use MongoDB for data persistence:

```typescript
// MongoDB Player repository
export class MongoPlayerRepository implements PlayerRepository {
  async getById(playerId: string): Promise<Player | undefined> {
    const playerDoc = await PlayerModel.findOne({ id: playerId });
    if (!playerDoc) return undefined;
    return this.playerMapper.toDomain(playerDoc);
  }

  async save(player: Player): Promise<void> {
    const playerData = this.playerMapper.toPersistence(player);
    await PlayerModel.findOneAndUpdate(
      { id: player.id },
      playerData,
      { upsert: true, new: true }
    );
  }
}
```

Implement offline progress calculation to process accumulated time when player returns:

```typescript
// Calculate offline progress when player logs back in
applyOfflineProgress(player: Player): Player {
  const now = Date.now();
  const elapsedTime = now - player.lastUpdate;
  
  // Cap offline progress (max 24 hours)
  const maxOfflineTime = 24 * 60 * 60 * 1000;
  const effectiveElapsedTime = Math.min(elapsedTime, maxOfflineTime);
  
  // Apply skill progress at a slightly reduced rate
  Object.values(player.skills).forEach(skill => {
    if (skill.isActive) {
      const offlineRate = 0.8; // 80% efficiency when offline
      const xpGain = calculateXpGain(skill, effectiveElapsedTime) * offlineRate;
      skill.addXp(xpGain);
      generateOfflineResources(player, skill, effectiveElapsedTime * offlineRate);
    }
  });
  
  return player;
}
```

- Create a satisfying progression curve—early levels should feel quick, later levels more challenging
- Balance resource generation rates with crafting requirements
- Design combat encounters with appropriate difficulty scaling
- Implement skill connections (e.g., Mining→Smithing→Combat) to create meaningful progression paths
